${NEX_TITLES[type] || 'Notice'}
${message}
`; container.appendChild(toast); requestAnimationFrame(() => requestAnimationFrame(() => toast.classList.add('show'))); // Auto close after 5s const t = setTimeout(() => dismissToast(toast), 5000); toast._timer = t;
